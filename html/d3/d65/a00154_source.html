29 void CommunityGenerator::Apply(shared_ptr<GeoGrid> geoGrid, const GeoGridConfig& geoGridConfig,
30  unsigned int& contactCenterCounter)
31 {
32  // 1. calculate number of communities, each community has average 2000 persons
33  // 2. assign communities to a location using a discrete distribution reflecting the relative number of
34  // people at that location
35 
36  const auto popCount = geoGridConfig.input.pop_size;
37  const auto communityCount =
38  static_cast<unsigned int>(ceil(popCount / geoGridConfig.constants.mean_community_size));
39 
40  vector<double> weights;
41  for (const auto& loc : *geoGrid) {
42  const auto weight = static_cast<double>(loc->GetPopCount()) / static_cast<double>(popCount);
43  CheckWeight("CommunityGenerator", weight);
44  weights.push_back(weight);
45  }
46 
47  if (weights.empty()) {
48  // trng can't handle empty vectors
49  return;
50  }
51 
52  const auto dist = m_rnManager[0].variate_generator(trng::discrete_dist(weights.begin(), weights.end()));
53 
54  for (auto i = 0U; i < communityCount; i++) {
55  const auto loc = (*geoGrid)[dist()];
56  const auto pc = make_shared<PrimaryCommunity>(contactCenterCounter++);
57  pc->Fill(geoGrid);
58  loc->AddContactCenter(pc);
59  }
60  for (auto i = 0U; i < communityCount; i++) {
61  const auto loc = (*geoGrid)[dist()];
62  const auto sc = make_shared<SecondaryCommunity>(contactCenterCounter++);
63  sc->Fill(geoGrid);
64  loc->AddContactCenter(sc);
65  }
66 }
