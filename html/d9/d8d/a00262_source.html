28 void K12SchoolPopulator::Apply(shared_ptr<GeoGrid> geoGrid, GeoGridConfig&)
29 {
30  m_logger->info("Starting to populate Schools");
31 
32  set<stride::ContactPool*> found;
33  unsigned int pupils = 0;
34 
35  // for every location
36  for (const shared_ptr<Location>& loc : *geoGrid) {
37  if (loc->GetPopulation() == 0) {
38  continue;
39  }
40 
41  // 1. find all schools in an area of 10-k*10 km
42  const vector<stride::ContactPool*>& classes = GetContactPoolInIncreasingRadius<K12School>(geoGrid, loc);
43 
44  auto dist = m_rnManager[0].variate_generator(
45  trng::uniform_int_dist(0, static_cast<trng::uniform_int_dist::result_type>(classes.size())));
46 
47  // 2. for every student assign a class
48  for (const shared_ptr<ContactCenter>& household : loc->GetContactCentersOfType<Household>()) {
49  stride::ContactPool* contactPool = household->GetPools()[0];
50  found.insert(contactPool);
51  for (stride::Person* p : *contactPool) {
52  if (p->IsStudentCandidate()) {
53  auto id = dist();
54  classes[id]->AddMember(p);
55  p->SetK12SchoolId(static_cast<unsigned int>(classes[id]->GetId()));
56  pupils++;
57  }
58  }
59  }
60  }
61 
62  m_logger->info("Number of pupils in schools: {}", pupils);
63  m_logger->info("Number of different classes: {}", found.size());
64 }
