219  void resize(size_type new_size)
220  {
221  if (new_size < size()) {
222  if (Safe) {
223  for (size_type i = m_size - 1; new_size - 1 < i; --i) {
224  pop_back();
225  }
226  } else {

228  while (new_block_count < get_block_count()) {
229  delete[] m_blocks[m_blocks.size() - 1];
230  m_blocks.pop_back();

233  }
