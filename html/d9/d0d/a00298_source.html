42 shared_ptr<Sim> SimBuilder::Build(shared_ptr<Sim> sim, shared_ptr<Population> pop)
43 {
44  // --------------------------------------------------------------
45  // Read config info and setup random number manager
46  // --------------------------------------------------------------
47  sim->m_config_pt = m_config_pt;
48  sim->m_population = std::move(pop);
49  sim->m_track_index_case = m_config_pt.get<bool>("run.track_index_case");
50  sim->m_adaptive_symptomatic_behavior = m_config_pt.get<bool>("run.adaptive_symptomatic_behavior", true);
51  sim->m_num_threads = m_config_pt.get<unsigned int>("run.num_threads");
52  sim->m_calendar = make_shared<Calendar>(m_config_pt);
53  sim->m_contact_log_mode = ContactLogMode::ToMode(m_config_pt.get<string>("run.contact_log_level", "None"));
54  sim->m_rn_manager.Initialize(
55  RnMan::Info{m_config_pt.get<string>("run.rng_seed", "1,2,3,4"), "", sim->m_num_threads});
56 
57  // --------------------------------------------------------------
58  // Contact handlers, each with generator bound to different
59  // random engine stream) and infector.
60  // --------------------------------------------------------------
61  for (size_t i = 0; i < sim->m_num_threads; i++) {
62  auto gen = sim->m_rn_manager[i].variate_generator(trng::uniform01_dist<double>());
63  sim->m_handlers.emplace_back(ContactHandler(gen));
64  }
65  const auto& select = make_tuple(sim->m_contact_log_mode, sim->m_track_index_case);
66  sim->m_infector = InfectorMap().at(select);
67 
68  // --------------------------------------------------------------
69  // Initialize the age-related contact profiles.
70  // --------------------------------------------------------------
71  const auto ageContactPt = ReadAgeContactPtree();
72  for (Id typ : IdList) {
73  sim->m_contact_profiles[typ] = AgeContactProfile(typ, ageContactPt);
74  }
75 
76  // --------------------------------------------------------------
77  // Initialize the transmission profile (fixes rates).
78  // --------------------------------------------------------------
79  const auto diseasePt = ReadDiseasePtree();
80  sim->m_transmission_profile.Initialize(m_config_pt, diseasePt);
81 
82  // --------------------------------------------------------------
83  // Initialize the public health agency (fixes detection probability).
84  // --------------------------------------------------------------
85  const double detection_probability = m_config_pt.get<double>("run.case_detection_probability", 0.0);
86  sim->m_public_health_agency.Initialize(detection_probability);
87 
88  // --------------------------------------------------------------
89  // Seed the population with health data.
90  // --------------------------------------------------------------
91  HealthSeeder(diseasePt).Seed(sim->m_population, sim->m_handlers);
92 
93  // --------------------------------------------------------------
94  // Seed population with immunity/vaccination/infection.
95  // --------------------------------------------------------------
96  DiseaseSeeder(m_config_pt, sim->m_rn_manager).Seed(sim->m_population);
97 
98  // --------------------------------------------------------------
99  // Seed population with survey participants.
100  // --------------------------------------------------------------
101  SurveySeeder(m_config_pt, sim->m_rn_manager).Seed(sim->m_population);
102 
103  // --------------------------------------------------------------
104  // Done.
105  // --------------------------------------------------------------
106  return sim;
107 }
