61 int Location::GetIncomingCommuterCount(double fractionCommuters) const
62 {
63  double value = 0;
64  for (const auto& locProportion : m_inCommuteLocations) {
65  // locProportion.second of the people in locProportion.first are commuting to this
66  value += locProportion.second * (fractionCommuters * (double)locProportion.first->GetPopCount());
67  }
68  return static_cast<int>(floor(value));
69 }

85 int Location::GetOutgoingCommuterCount(double fractionCommuters) const
86 {
