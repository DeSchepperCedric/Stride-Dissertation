44 
45  auto ref_k12school_age = 0U;

47  auto ref_workplace_age = 0U;

53  if (College::HasAge(age)) {

55  }
56  if (Workplace::HasAge(age)) {
57  ref_workplace_age++;
58  }
59  }

63  //----------------------------------------------------------------
64  const auto fraction_k12school_age = static_cast<double>(ref_k12school_age) / static_cast<double>(ref_pop_count);
65  const auto fraction_college_age = static_cast<double>(ref_college_age) / static_cast<double>(ref_pop_count);
66  const auto fraction_workplace_age = static_cast<double>(ref_workplace_age) / static_cast<double>(ref_pop_count);
67

69  const auto age_count_college = static_cast<unsigned int>(floor(popSize * fraction_college_age));
70  const auto age_count_workplace = static_cast<unsigned int>(floor(popSize * fraction_workplace_age));

76

79  (age_count_workplace - geoGridConfig.popInfo.popcount_college)));
