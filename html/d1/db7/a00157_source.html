144 vector<unsigned int> RunConfigManager::CreateNumThreads(unsigned int max)
145 {
146  max = std::max(max, ConfigInfo::NumberAvailableThreads());
147  initializer_list<unsigned int> num{1U};
148 
149  if (max >= 2) {
150  num = {1U, 2U};
151  }
152  if (max >= 4) {
153  num = {1U, 2U, 3U, 4U};
154  }
155  if (max >= 8) {
156  num = {1U, 2U, 3U, 4U, 5U, 6U, 7U, 8U};
157  }
158  if (max >= 12) {
159  num = {1U, 2U, 3U, 4U, 5U, 6U, 7U, 8U, 10U, 12U};
160  }
161  if (max >= 16) {
162  num = {1U, 2U, 3U, 4U, 5U, 6U, 7U, 8U, 10U, 12U, 16U};
163  }
164 
165  return num;
166 }
