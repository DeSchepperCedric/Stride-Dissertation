158 void MyhayaiController::RunBenchmarks()
159 {
160  BenchmarkRunner& runner = BenchmarkRunner::Instance();
161  BenchControlHelper helper(runner.GetTestDescriptors(), m_negative, m_positive);
162 
163  // Get the the canonical test names, after applying include/exclude regex and shuffle.
164  auto names = helper.GetIncludedNames();
165  if (m_shuffle) {
166  helper.Shuffle(names);
167  }
168 
169  // Always the console viewer.
170  using cv_t = ConsoleViewer<chrono::milliseconds>;
171  auto cv = make_shared<cv_t>(cout, m_no_color);
172  runner.Register(cv, bind(&cv_t::Update, cv, placeholders::_1));
173 
174  // Possibly the ptree viewer.
175  using pv_t = PtreeViewer<chrono::milliseconds>;
176  auto pv = make_shared<pv_t>();
177  if (!(m_info_path.empty() && m_json_path.empty() && m_xml_path.empty())) {
178  runner.Register(pv, bind(&pv_t::Update, pv, placeholders::_1));
179  }
180 
181  // Run them.
182  runner.RunTests(names);
183 
184  // Possibly the ptree viewer.
185  if (!m_info_path.empty()) {
186  write_info(m_info_path, pv->CGet());
187  }
188  if (!m_xml_path.empty()) {
189  write_json(m_xml_path, pv->CGet());
190  }
191  if (!m_xml_path.empty()) {
192  write_xml(m_xml_path, pv->CGet(), std::locale(), xml_writer_make_settings<ptree::key_type>(' ', 8));
193  }
194 }
