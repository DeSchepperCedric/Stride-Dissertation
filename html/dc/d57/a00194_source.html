63 void RnPcg::Seed(const Info& info)
64 {
65  if (m_stream_count != info.m_stream_count) {
66  throw runtime_error("RnManager> stream count does not match in Seed.");
67  }
68  m_seed_seq_init = info.m_seed_seq_init;
69 
70  auto state = info.m_state;
71  if (state.empty()) {
72  vector<string> seed_vec = Split(m_seed_seq_init, ",");
73  vector<unsigned int> seed_seq_vec;
74  for (const auto& e : seed_vec) {
75  if (!CheckAllDigits(e)) {
76  throw runtime_error("RnManager> Error in seeding definiton.");
77  }
78  seed_seq_vec.push_back(FromString<unsigned int>(e));
79  }
80 
81  seed_seq_fe128 seseq(seed_seq_vec.begin(), seed_seq_vec.end());
82  vector<pcg64::state_type> seed(2*m_stream_count);
83  seseq.generate(seed.begin(), seed.end());
84 
85  for (size_t i = 0; i < m_stream_count; ++i) {
86  m_engines[i].seed(seed[i], seed[i+1]);
87  }
88  } else {
89  stringstream ss(state);
90  for (size_t i = 0; i < m_stream_count; ++i) {
91  ss >> m_engines[i];
92  }
