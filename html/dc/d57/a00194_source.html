62 std::map<std::string, std::vector<std::vector<unsigned int>>> Calibrator::RunMultiple(
63  unsigned int count, const std::vector<std::pair<boost::property_tree::ptree, std::string>>& configs) const
64 {
65  std::map<std::string, std::vector<std::vector<unsigned int>>> results;
66 
67  std::random_device rd;
68 
69  std::vector<std::random_device::result_type> seeds;
70 
71  for (unsigned int i = 0; i < count * configs.size(); i++) {
72  seeds.push_back(rd());
73  }
74 
75  for (const auto& config : configs) {
76  results[config.second].resize(count);
77  }
78 #pragma omp parallel for collapse(2)
79  for (unsigned int config = 0; config < configs.size(); config++) {
80  for (unsigned int i = 0; i < count; i++) {
81  auto tag = configs[config].second;
82  auto config_pt = configs[config].first;
83  auto seed = seeds[config * count + i];
84 
85  config_pt.put("run.rng_seed", seed);
86  logger->info("Starting the testcase {}, run {} of {} using seed {}", tag, i, count, seed);
87  stride::util::RnMan rn_manager;
88  rn_manager.Initialize(
89  stride::util::RnMan::Info{config_pt.get<std::string>("run.rng_seed", "1,2,3,4"), "",
90  config_pt.get<unsigned int>("run.num_threads")});
91  auto pop = stride::Population::Create(config_pt, rn_manager);
92  auto runner = std::make_shared<stride::SimRunner>(config_pt, pop, rn_manager);
93  std::shared_ptr<stride::viewers::InfectedViewer> infectedViewer =
94  std::make_shared<stride::viewers::InfectedViewer>(runner);
95  runner->Register(infectedViewer, bind(&stride::viewers::InfectedViewer::Update, infectedViewer,
96  std::placeholders::_1));
97  runner->Run();
98 
99  // Get the infected count
100  const unsigned int res = runner->GetSim()->GetPopulation()->GetInfectedCount();
101  logger->info("Finished running testcase {}, {} people were infected", tag, res);
102  results[tag][i] = infectedViewer->GetInfectionCounts();
103  }
104  }
105  return results;
106 }

115 std::pair<double, double> Calibrator::FindMeanStdev(std::vector<unsigned int> results) const
116 {
117  double sum = std::accumulate(results.begin(), results.end(), 0.0);
118  double mean = sum / results.size();
119 
120  // Inspired by https://stackoverflow.com/a/7616783
121  std::vector<double> diff(results.size());
122  std::transform(results.begin(), results.end(), diff.begin(), [mean](double x) { return x - mean; });
123  double sq_sum = std::inner_product(diff.begin(), diff.end(), diff.begin(), 0.0);
124  double stdev = std::sqrt(sq_sum / results.size());
125  return std::make_pair(mean, stdev);
126 }

175 void Calibrator::WriteResults(const std::map<std::string, std::vector<unsigned int>>& single,
176  const std::map<std::string, std::vector<std::vector<unsigned int>>>& multiple,
177  std::string filename) const
178 {
179  boost::property_tree::ptree root;
180  std::set<std::string> tags;
181 
182  for (const auto& tag : single)
183  tags.insert(tag.first);
184  for (const auto& tag : multiple)
185  tags.insert(tag.first);
186 
187  for (const auto& tag : tags) {
188  boost::property_tree::ptree tags_root;
189  unsigned int size;
190  if (multiple.count(tag) && !multiple.at(tag).empty())
191  size = multiple.at(tag)[0].size();
192  else if (single.count(tag) && !single.at(tag).empty())
193  size = single.at(tag).size();
194  else
195  continue;
196  for (unsigned int step = 0; step < size; step++) {
197  boost::property_tree::ptree step_root;
198  if (multiple.count(tag)) {
199  std::vector<unsigned int> data;
200  boost::property_tree::ptree results;
201  for (const auto& vect : multiple.at(tag)) {
202  if (vect.size() - 1 < step)
203  continue;
204  data.push_back(vect[step]);
205  boost::property_tree::ptree result;
206  result.put("", vect[step]);
207  results.push_back(std::make_pair("", result));
208  }
209  auto p = FindMeanStdev(data);
210  step_root.put("mean", p.first);
211  step_root.put("standard deviation", p.second);
212  step_root.add_child("data", results);
213  }
214  if (single.count(tag)) {
215  step_root.put("exact", single.at(tag)[step]);
216  }
217  tags_root.add_child(std::to_string(step), step_root);
218  }
219  root.add_child(tag, tags_root);
220  }
221  std::ofstream file;
222  file.open(filename);
223  if (!file.good()) {
224  logger->error("There was a problem creating the file '{}'", filename);
225  return;
226  }
227  boost::property_tree::write_json(file, root);
228 }
