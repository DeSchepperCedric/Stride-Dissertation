30 void CollegePopulator::Apply(std::shared_ptr<GeoGrid> geoGrid, GeoGridConfig& geoGridConfig)
31 {
32  m_logger->info("Starting to populate Colleges");
33 
34  std::set<stride::ContactPool*> found;
35  unsigned int students = 0;
36  unsigned int commuting = 0;
37 
38  // for every location
39  for (const std::shared_ptr<Location>& loc : *geoGrid) {
40  if (loc->GetPopulation() == 0) {
41  continue;
42  }
43  // 1. find all highschools in an area of 10-k*10 km
44  const std::vector<stride::ContactPool*>& nearByHighSchools =
45  GetContactPoolInIncreasingRadius<College>(geoGrid, loc);
46 
47  ExcAssert(!nearByHighSchools.empty(),
48  "Did not find any HighSchool due to invalid input data in HighSchoolPopulator");
49 
50  auto distNonCommuting = m_rnManager[0].variate_generator(trng::uniform_int_dist(
51  0, static_cast<trng::uniform_int_dist::result_type>(nearByHighSchools.size())));
52 
53  // 2. find all highschools were students from this location commute to
54  std::vector<Location*> commutingHighSchools;
55  std::vector<double> commutingWeights;
56  for (const std::pair<Location*, double>& commute : loc->GetOutgoingCommuningCities()) {
57  const auto& highSchools = commute.first->GetContactCentersOfType<College>();
58  if (!highSchools.empty()) {
59  commutingHighSchools.push_back(commute.first);
60  commutingWeights.push_back(commute.second);
61  }
62  }
63 
64  std::function<trng::discrete_dist::result_type()> disCommuting;
65 
66  if (!commutingWeights.empty()) {
67  disCommuting = m_rnManager[0].variate_generator(
68  trng::discrete_dist(commutingWeights.begin(), commutingWeights.end()));
69  }
70 
71  // 2. for every student assign a class
72  for (const std::shared_ptr<ContactCenter>& household : loc->GetContactCentersOfType<Household>()) {
73  stride::ContactPool* contactPool = household->GetPools()[0];
74  found.insert(contactPool);
75  for (stride::Person* person : *contactPool) {
76  if (person->IsCollegeStudentCandidate() &&
77  MakeChoice(geoGridConfig.input.fraction_1826_years_WhichAreStudents)) {
78  students++;
79  // this person is a student
80  if (!commutingHighSchools.empty() &&
81  MakeChoice(geoGridConfig.input.fraction_student_commutingPeople)) {
82  // this person is commuting
83  commuting++;
84 
85  // id of the location this person is commuting to
86  auto locationId = disCommuting();
87  // create a list of ContactPools (i.e. classes) for each of highschool
88  // of this location
89  const auto& highSchools = commutingHighSchools[locationId]
90  ->GetContactCentersOfType<College>();
91 
92  std::vector<stride::ContactPool*> contactPools;
93  for (const auto& highSchool : highSchools) {
94  contactPools.insert(contactPools.end(), highSchool->begin(),
95  highSchool->end());
96  }
97 
98  auto disPools = m_rnManager[0].variate_generator(trng::uniform_int_dist(
99  0, static_cast<trng::uniform_int_dist::result_type>(
100  contactPools.size())));
101 
102  auto id = disPools();
103  contactPools[id]->AddMember(person);
104  person->SetCollegeId(contactPools[id]->GetId());
105  } else {
106  auto id = distNonCommuting();
107  nearByHighSchools[id]->AddMember(person);
108  person->SetCollegeId(nearByHighSchools[id]->GetId());
109  }
110  }
111  }
112  }
113  }
114  m_logger->info("Finished populating Colleges");
115  m_logger->info("Used {} different classes", found.size());
116  m_logger->info("Placed {} students in Colleges, {} of whom are commuting to a different location", students,
117  commuting);
118 }
