31 void GeoGridProtoWriter::Write(shared_ptr<gengeopop::GeoGrid> geoGrid, ostream& stream)
32 {
33  GOOGLE_PROTOBUF_VERIFY_VERSION;
34 
35  proto::GeoGrid protoGrid;
36  for (const auto& location : *geoGrid) {
37  WriteLocation(location, protoGrid.add_locations());
38  }
39  for (const auto& person : m_persons_found) {
40  WritePerson(person, protoGrid.add_persons());
41  }
42 
43  m_persons_found.clear();
44  if (!protoGrid.SerializeToOstream(&stream)) {
45  throw stride::util::Exception("There was an error writing the GeoGrid to the file.");
46  }
47  google::protobuf::ShutdownProtobufLibrary();
48  stream.flush();
49 }
