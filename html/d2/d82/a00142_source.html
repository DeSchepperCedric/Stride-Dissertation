39 shared_ptr<Population> PopBuilder::MakePoolSys(std::shared_ptr<Population> pop)
40 {
41  using namespace ContactPoolType;
42  auto& population = *pop;
43  auto& poolSys = population.GetContactPoolSys();
44 
45  // --------------------------------------------------------------
46  // Determine maximum pool ids in population.
47  // --------------------------------------------------------------
48  IdSubscriptArray<unsigned int> max_ids{0U};
49  for (const auto& p : population) {
50  for (Id typ : IdList) {
51  max_ids[typ] = max(max_ids[typ], p.GetPoolId(typ));
52  }
53  }
54  // --------------------------------------------------------------
55  // Initialize poolSys with empty ContactPools (even for Id=0).
56  // --------------------------------------------------------------
57  for (Id typ : IdList) {
58  for (size_t i = 0; i < max_ids[typ] + 1; i++) {
59  poolSys[typ].emplace_back(ContactPool(i, typ));
60  }
61  }
62 
63  // --------------------------------------------------------------
64  // Insert persons (pointers) in their contactpools. Having Id 0
65  // means "not belonging pool of that type" (e.g. school/ work -
66  // cannot belong to both, or e.g. out-of-work).
67  //
68  // Pools are uniquely identified by (typ, subscript) and a Person
69  // belongs, for typ, to pool with subscrip p.GetPoolId(typ).
70  // Defensive measure: we have a pool for Id 0 and leave it empty.
71  // --------------------------------------------------------------
72  for (auto& p : population) {
73  for (Id typ : IdList) {
74  const auto poolId = p.GetPoolId(typ);
75  if (poolId > 0) {
76  poolSys[typ][poolId].AddMember(&p);
77  }
78  }
79  }
80 
81  return pop;
82 }

84 shared_ptr<Population> PopBuilder::MakePersons(std::shared_ptr<Population> pop)
85 {
86  //------------------------------------------------
87  // Read persosns from file.
88  //------------------------------------------------
89  const auto file_name = m_config_pt.get<string>("run.population_file");
90  const auto use_install_dirs = m_config_pt.get<bool>("run.use_install_dirs");
91  const auto file_path = (use_install_dirs) ? FileSys::GetDataDir() /= file_name : file_name;
92  if (!is_regular_file(file_path)) {
93  throw runtime_error(string(__func__) + "> Population file " + file_path.string() + " not present.");
94  }
95 
96  boost::filesystem::ifstream pop_file;
97  pop_file.open(file_path.string());
98  if (!pop_file.is_open()) {
99  throw runtime_error(string(__func__) + "> Error opening population file " + file_path.string());
100  }
101 
102  string line;
103  getline(pop_file, line); // step over file header
104  unsigned int person_id = 0U;
105 
106  while (getline(pop_file, line)) {
107  const auto values = Split(line, ",");
108  const auto age = FromString<unsigned int>(values[0]);
109  const auto household_id = FromString<unsigned int>(values[1]);
110  const auto school_id = FromString<unsigned int>(values[2]);
111  const auto work_id = FromString<unsigned int>(values[3]);
112  const auto primary_community_id = FromString<unsigned int>(values[4]);
113  const auto secondary_community_id = FromString<unsigned int>(values[5]);
114 
115  pop->CreatePerson(person_id, age, household_id, school_id, work_id, primary_community_id,
116  secondary_community_id);
117  ++person_id;
118  }
119 
120  pop_file.close();
121  return pop;
122 }
