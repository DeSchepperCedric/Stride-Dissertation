29  const auto pupilCount = geoGridConfig.popInfo.popcount_1826_student;
30  const auto schoolCount =
31  static_cast<unsigned int>(ceil(pupilCount / geoGridConfig.constants.mean_college_size));
32  const auto cities = geoGrid->TopK(10);
33 
34  if (cities.empty()) {
35  // trng can't handle empty vectors
36  return;
37  }
38 
39  // Aggregate population in cities.
40  auto totalPop = 0U;
41  for (const shared_ptr<Location>& c : cities) {
42  totalPop += c->GetPopCount();
43  }
44 
45  // Weights determined by relative population in city.
46  vector<double> weights;
47  for (const auto& c : cities) {

49  CheckWeight("CollegeGenerator", weight);
50  weights.push_back(weight);
51  }
52 
53  const auto dist = m_rnManager[0].variate_generator(trng::discrete_dist(weights.begin(), weights.end()));
54
