46 std::vector<std::shared_ptr<Location>> GeoGrid::TopK(size_t k) const
47 {
48  auto cmp = [](const std::shared_ptr<Location>& rhs, const std::shared_ptr<Location>& lhs) {
49  return rhs->GetPopulation() > lhs->GetPopulation();
50  };
51 
52  std::priority_queue<std::shared_ptr<Location>, std::vector<std::shared_ptr<Location>>, decltype(cmp)> queue(
53  cmp);
54 
55  for (auto it = cbegin(); it != cend(); it++) {
56  queue.push(*it);
57  if (queue.size() > k) {
58  queue.pop();
59  }
60  }
61 
62  std::vector<std::shared_ptr<Location>> topLocations;
63 
64  while (!queue.empty()) {
65  auto loc = queue.top();
66  topLocations.push_back(loc);
67  queue.pop();
68  }
69 
70  return topLocations;
71 }
