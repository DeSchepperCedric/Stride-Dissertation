130  RadiusPolicy(Args args) : m_center(std::move(std::get<0>(args))), m_radius(std::get<1>(args)) {}
131 
132  AABB<geogrid_detail::KdTree2DPoint> GetBoundingBox() const
133  {
134  using namespace geoaggregator_detail;
135 
136  AABB<geogrid_detail::KdTree2DPoint> box{};
137 
138  // As of boost 1.66, there's seems no way to do this in Boost.Geometry
139  constexpr double EARTH_RADIUS_KM = 6371.0;
140  double scaled_radius = m_radius / EARTH_RADIUS_KM;
141 
142  double startlon = m_center.Get<0>();
143  double startlat = m_center.Get<1>();
144  double londiff = RadianToDegree(scaled_radius / std::cos(DegreeToRadian(startlat)));
145  double latdiff = RadianToDegree(scaled_radius);
146 
147  box.upper = geogrid_detail::KdTree2DPoint(startlon + londiff, startlat + latdiff);
148  box.lower = geogrid_detail::KdTree2DPoint(startlon - londiff, startlat - latdiff);
149 
150  return box;
151  }
152 
153  bool Contains(const geogrid_detail::KdTree2DPoint& pt) const { return pt.InRadius(m_center, m_radius); }
154 
155 private:
