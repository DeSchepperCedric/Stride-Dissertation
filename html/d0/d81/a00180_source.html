50 void BenchmarkRunner::RunTests(const vector<string>& canonicalNames)
51 {
52  // Setup.
53  auto test_descriptors = Instance().GetTestDescriptors();
54  Notify(event::Payload(event::Id::BeginBench));
55 
56  // Run through all the test_descriptors in ascending order.
57  for (const auto& name : canonicalNames) {
58  // If test is disabled, then skip it.
59  const auto& t_d = test_descriptors[name];
60  if (t_d.m_is_disabled) {
61  Notify(event::Payload(event::Id::SkipTest, name));
62  continue;
63  }
64 
65  // Runs for test with canonical name <name>.
66  Notify(event::Payload(event::Id::BeginTest, name));
67  TestResult runTimes(t_d.m_num_runs);
68  try {
69  for (auto& rt : runTimes) {
70  rt = t_d.m_test_factory().Run<steady_clock>();
71  }
72  } catch (exception& e) {
73  Notify(event::Payload(event::Id::AbortTest, name, e.what()));
74  continue;
75  }
76 
77  // Runs for this test done.
78  Notify(event::Payload(event::Id::EndTest, name, runTimes));
79  }
80 
81  // End output.
82  Notify(event::Payload(event::Id::EndBench));
83 }
