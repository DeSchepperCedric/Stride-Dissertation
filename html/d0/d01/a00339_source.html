44 void WorkplacePopulator::Apply(shared_ptr<GeoGrid> geoGrid, const GeoGridConfig& geoGridConfig)
45 {
46  m_logger->info("Starting to populate Workplaces");
47 
48  m_geoGrid = geoGrid;
49  m_geoGridConfig = geoGridConfig;
50  m_fractionCommutingStudents = 0;
51  m_workplacesInCity.clear();
52  m_currentLoc = nullptr;
53  m_assignedTo0 = 0;
54  m_assignedCommuting = 0;
55  m_assignedNotCommuting = 0;
56  m_distNonCommuting = discreteDist();
57  m_nearByWorkplaces.clear();
58  m_disCommuting = discreteDist();
59  m_commutingLocations.clear();
60 
61  CalculateFractionCommutingStudents();
62  CalculateWorkplacesInCity();
63 
64  // for every location
65  for (const auto& loc : *geoGrid) {
66  if (loc->GetPopCount() == 0) {
67  continue;
68  }
69  m_currentLoc = loc;
70  CalculateCommutingLocations();
71  CalculateNearbyWorkspaces();
72 
73  // 2. for everyone of working age: decide between work or college (iff of College age)
74  for (const auto& household : loc->GetContactCentersOfType<Household>()) {
75  auto contactPool = household->GetPools()[0];
76  for (auto p : *contactPool) {
77  if (AgeBrackets::Workplace::HasAge((p->GetAge()))) {
78  bool isStudent = MakeChoice(geoGridConfig.input.participation_college);
79  bool isActiveWorker = MakeChoice(geoGridConfig.input.particpation_workplace);
80 
81  if ((AgeBrackets::College::HasAge(p->GetAge()) && !isStudent) ||
82  isActiveWorker) {
83  AssignActive(p);
84  } else {
85  // this person has no employment
86  p->SetPoolId(Id::Workplace, 0);
87  m_assignedTo0++;
88  }
89  }
90  }
91  }
92  }
93 
94  m_logger->info("Populated workplaces, assigned to 0 {}, assigned (commuting) {} assigned (not commuting) {} ",
95  m_assignedTo0, m_assignedCommuting, m_assignedNotCommuting);
96 }

98 void WorkplacePopulator::CalculateFractionCommutingStudents()
99 {
100  m_fractionCommutingStudents = 0;
101  if (static_cast<bool>(m_geoGridConfig.input.fraction_workplace_commuters) &&
102  m_geoGridConfig.popInfo.popcount_workplace) {
103  m_fractionCommutingStudents =
104  (m_geoGridConfig.popInfo.popcount_college * m_geoGridConfig.input.fraction_college_commuters) /
105  (m_geoGridConfig.popInfo.popcount_workplace * m_geoGridConfig.input.fraction_workplace_commuters);
106  }

109 void WorkplacePopulator::CalculateWorkplacesInCity()
110 {
111  for (const shared_ptr<Location>& loc : *m_geoGrid) {
112  vector<ContactPool*> contactPools;
113  for (const auto& wp : loc->GetContactCentersOfType<Workplace>()) {
114  contactPools.insert(contactPools.end(), wp->begin(), wp->end());
115  }
116 
117  auto disPools = m_rnManager[0].variate_generator(
118  trng::uniform_int_dist(0, static_cast<trng::uniform_int_dist::result_type>(contactPools.size())));
119 
120  m_workplacesInCity[loc.get()] = {contactPools, disPools};
121  }
122 }

124 void WorkplacePopulator::AssignActive(Person* person)
125 {
126  // this person is (student and active) or active
127  if (!m_commutingLocations.empty() && MakeChoice(m_geoGridConfig.input.fraction_workplace_commuters)) {
128  // this person is commuting
129  const auto& info = m_workplacesInCity[m_commutingLocations[m_disCommuting()]];
130  const auto id = info.second(); // id of the location this person is commuting to
131  info.first[id]->AddMember(person);
132  person->SetPoolId(Id::Workplace, info.first[id]->GetId());
133  m_assignedCommuting++;
134  } else {
135  // this person is not commuting
136  const auto id = m_distNonCommuting();
137  m_nearByWorkplaces[id]->AddMember(person);
138  person->SetPoolId(Id::Workplace, m_nearByWorkplaces[id]->GetId());
139  m_assignedNotCommuting++;
140  }
141 }

143 void WorkplacePopulator::CalculateCommutingLocations()
144 {
145  // find all Workplaces were employees from this location commute to
146  m_commutingLocations.clear();
147  m_disCommuting = discreteDist();
148 
149  vector<double> commutingWeights;
150  for (const pair<Location*, double>& commute : m_currentLoc->GetOutgoingCommutingCities()) {
151  const auto& workplaces = commute.first->GetContactCentersOfType<Workplace>();
152  if (!workplaces.empty()) {
153  m_commutingLocations.push_back(commute.first);
154  const auto weight = commute.second - (commute.second * m_fractionCommutingStudents);
155  commutingWeights.push_back(weight);
156  ExcAssert(weight >= 0 && weight <= 1 && !isnan(weight),
157  "Invalid weight due to data in WorkplacePopulator, weight: " + to_string(weight));
158  }
159  }
160 
161  if (!commutingWeights.empty()) {
162  m_disCommuting = m_rnManager[0].variate_generator(
163  trng::discrete_dist(commutingWeights.begin(), commutingWeights.end()));
164  }
165 }
