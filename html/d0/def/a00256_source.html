49 void GenGeoPopController::ReadDataFiles()
50 {
51  m_geoGrid = std::make_shared<GeoGrid>(m_population.get());
52 
53  ReaderFactory readerFactory;
54 
55 #pragma omp parallel sections
56  {
57 #pragma omp section
58  {
59  m_citiesReader = readerFactory.CreateCitiesReader(m_citiesFileName);
60  m_citiesReader->FillGeoGrid(m_geoGrid);
61  }
62 
63 #pragma omp section
64  {
65  if (m_commutingFileName != "") {
66  m_commutesReader = readerFactory.CreateCommutesReader(m_commutingFileName);
67  }
68  }
69 
70 #pragma omp section
71  {
72  m_householdsReader = readerFactory.CreateHouseholdReader(m_householdsFileName);
73  }
74  }
75 
76  if (m_commutingFileName != "") {
77  m_commutesReader->FillGeoGrid(m_geoGrid);
78  }
79 
80  m_geoGridConfig.Calculate(m_geoGrid, m_householdsReader);
81 
82  m_geoGrid->Finalize();
83 }
