49 void GenPopController::ReadDataFiles(const std::string& citiesFileName, const std::string& commutingFileName,
50  const std::string& householdsFileName)
51 {
52  m_geoGrid = make_shared<GeoGrid>(m_population.get());
53 
54  ReaderFactory readerFactory;
55 
56 #pragma omp parallel sections
57  {
58 #pragma omp section
59  {
60  m_citiesReader = readerFactory.CreateCitiesReader(citiesFileName);
61  m_citiesReader->FillGeoGrid(m_geoGrid);
62  }
63 
64 #pragma omp section
65  {
66  if (!commutingFileName.empty()) {
67  m_commutesReader = readerFactory.CreateCommutesReader(commutingFileName);
68  }
69  }
70 
71 #pragma omp section
72  {
73  m_householdsReader = readerFactory.CreateHouseholdReader(householdsFileName);
74  }
75  }
76 
77  if (!commutingFileName.empty()) {
78  m_commutesReader->FillGeoGrid(m_geoGrid);
79  }
80 
81  m_geoGridConfig.Calculate(m_geoGrid, m_householdsReader);
82  m_geoGrid->Finalize();
83 }
