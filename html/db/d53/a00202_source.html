49 vector<shared_ptr<Location>> GeoGrid::TopK(size_t k) const
50 {
51  auto cmp = [](const shared_ptr<Location>& rhs, const shared_ptr<Location>& lhs) {
52  return rhs->GetPopulation() > lhs->GetPopulation();
53  };
54 
55  priority_queue<shared_ptr<Location>, vector<shared_ptr<Location>>, decltype(cmp)> queue(
56  cmp);
57 
58  for (auto it = cbegin(); it != cend(); it++) {
59  queue.push(*it);
60  if (queue.size() > k) {
61  queue.pop();
62  }
63  }
64 
65  vector<shared_ptr<Location>> topLocations;
66 
67  while (!queue.empty()) {
68  auto loc = queue.top();
69  topLocations.push_back(loc);
70  queue.pop();
71  }
72 
73  return topLocations;
74 }
