66 inline std::vector<std::string> Tokenize(const std::string& str, const std::string& delimiters)
67 {
68  std::vector<std::string> tokens;
69 
70  // Skip delimiters at beginning.
71  std::string::size_type lastPos = str.find_first_not_of(delimiters, 0);
72  // Find first non-delimiter.
73  std::string::size_type pos = str.find_first_of(delimiters, lastPos);
74 
75  while (std::string::npos != pos || std::string::npos != lastPos) {
76  // Found a token, add it to the vector.
77  tokens.push_back(str.substr(lastPos, pos - lastPos));
78  // Skip delimiters.
79  lastPos = str.find_first_not_of(delimiters, pos);
80  // Find next non-delimiter.
81  pos = str.find_first_of(delimiters, lastPos);
82  }
83 
84  return tokens;
85 }
