38 shared_ptr<Population> PopBuilder::MakePoolSys(std::shared_ptr<Population> pop)
39 {
40  using namespace ContactPoolType;
41  auto& population = *pop;
42  auto& poolSys = population.GetContactPoolSys();
43 
44  // --------------------------------------------------------------
45  // Determine maximum pool ids in population.
46  // --------------------------------------------------------------
47  IdSubscriptArray<unsigned int> max_ids{0U};
48  for (const auto& p : population) {
49  for (Id typ : IdList) {
50  max_ids[typ] = max(max_ids[typ], p.GetPoolId(typ));
51  }
52  }
53  // --------------------------------------------------------------
54  // Initialize poolSys with empty ContactPools (even for Id=0).
55  // --------------------------------------------------------------
56  for (Id typ : IdList) {
57  for (size_t i = 0; i < max_ids[typ] + 1; i++) {
58  poolSys[typ].emplace_back(ContactPool(i, typ));
59  }
60  }
61 
62  // --------------------------------------------------------------
63  // Insert persons (pointers) in their contactpools. Having Id 0
64  // means "not belonging pool of that type" (e.g. school/ work -
65  // cannot belong to both, or e.g. out-of-work).
66  //
67  // Pools are uniquely identified by (typ, subscript) and a Person
68  // belongs, for typ, to pool with subscrip p.GetPoolId(typ).
69  // Defensive measure: we have a pool for Id 0 and leave it empty.
70  // --------------------------------------------------------------
71  for (auto& p : population) {
72  for (Id typ : IdList) {
73  const auto poolId = p.GetPoolId(typ);
74  if (poolId > 0) {
75  poolSys[typ][poolId].AddMember(&p);
76  }
77  }
78  }
79 
80  return pop;
81 }

83 shared_ptr<Population> PopBuilder::MakePersons(std::shared_ptr<Population> pop)
84 {
85  //------------------------------------------------
86  // Read persons from file.
87  //------------------------------------------------
88  const auto file_name = m_config_pt.get<string>("run.population_file");
89  const auto use_install_dirs = m_config_pt.get<bool>("run.use_install_dirs");
90  const auto file_path = (use_install_dirs) ? FileSys::GetDataDir() /= file_name : file_name;
91  if (!is_regular_file(file_path)) {
92  throw runtime_error(string(__func__) + "> Population file " + file_path.string() + " not present.");
93  }
94 
95  boost::filesystem::ifstream pop_file;
96  pop_file.open(file_path.string());
97  if (!pop_file.is_open()) {
98  throw runtime_error(string(__func__) + "> Error opening population file " + file_path.string());
99  }
100 
101  string line;
102  getline(pop_file, line); // step over file header
103  unsigned int person_id = 0U;
104 
105  while (getline(pop_file, line)) {
106  const auto values = Split(line, ",");
107  const auto age = FromString<unsigned int>(values[0]);
108  const auto household_id = FromString<unsigned int>(values[1]);
109  const auto school_id = FromString<unsigned int>(values[2]);
110  const auto work_id = FromString<unsigned int>(values[3]);
111  const auto primary_community_id = FromString<unsigned int>(values[4]);
112  const auto secondary_community_id = FromString<unsigned int>(values[5]);
113 
114  pop->CreatePerson(person_id, age, household_id, school_id, work_id, primary_community_id,
115  secondary_community_id);
116  ++person_id;
117  }
118 
119  pop_file.close();
120  return pop;
121 }
